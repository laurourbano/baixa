function calculaDesconto() {

    let litros;
    let valor_gasolina = 3.3;
    let valor_alcool = 2.9;
    let total;
    let desconto;
    let total_apos_desconto;

    litros = window.prompt("digite quantos litros foram abastecidos: ");
    console.log("você abasteceu " + litros + " litros");

    let combustivel = window.prompt("digite 1 para gasolina e 2 para alcool: ");

    	switch (combustivel) {
        	case "1":
            	total = litros * valor_gasolina;

	if (litros <= 20) {
                	desconto = total * 0.03;
                	total_apos_desconto = total - desconto;
                	console.log(
                    "Como foram abastecidos menos de 20 litros de gasolina, você teve um desconto de: R$" + desconto.toFixed(2)
                	);
                	console.log(
                    "Com isso você irá pagar R$" + total_apos_desconto.toFixed(2)
                	);
            	} else {
                	desconto = total * 0.05;
                	total_apos_desconto = total - desconto;
                	console.log(
                    "Como foram abastecidos mais de 20 litros de gasolina, você teve um desconto de: R$" + desconto.toFixed(2)
                	);
                	console.log(
                    "Com isso você irá pagar R$" + total_apos_desconto.toFixed(2)
                	);
            	}
            	break;
        	case "2":
            	total = litros * valor_alcool;
            	if (litros <= 20) {
                	desconto = total * 0.04;
                	total_apos_desconto = total - desconto;
                	console.log(
                    "Como foram abastecidos menos de 20 litros de alcool, você teve um desconto de: R$" + desconto.toFixed(2)
                	);
                	console.log(
                    "Com isso você irá pagar R$" + total_apos_desconto.toFixed(2)
                	);
            	} else {
                	desconto = total * 0.06;
                	total_apos_desconto = total - desconto;
                	console.log(
                    "Como foram abastecidos mais de 20 litros de alcool, você teve um desconto de: R$" + desconto.toFixed(2)
                	);
                	console.log(
                    "Com isso você irá pagar R$" + total_apos_desconto.toFixed(2)
                	);
            	}
            	break;
    	}
}
